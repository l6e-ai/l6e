1"""Cloud-sync configuration and ``/v1/authorize`` client for the SDK. 

2 

3Opt-in surface for the cloud-sync integration tier. A customer wires it 

4via the ``cloud=`` kwarg on ``pipeline()``; absent that kwarg, the SDK 

5behaves exactly as it did before this module existed (purely local 

6enforcement, no network calls). 

7 

8This module owns three responsibilities: 

9 

101. ``CloudConfig`` — the user-facing configuration surface. 

112. ``_post_authorize`` — the synchronous HTTP client for 

12 ``POST /v1/authorize``. Returns ``None`` on every failure path 

13 (timeout, non-2xx, malformed JSON, network exception, missing API 

14 key, garbage envelope) so the caller can fall back to local 

15 enforcement. The gate fails open, always — that's the iron rule. 

163. ``_sanitize_authorize_response`` — defensive validation of the server 

17 envelope before any field is used to drive a ``GateDecision``. NaN / 

18 negative / missing-action / unknown-pressure-label all collapse to 

19 ``None`` so a poisoned response can't quietly corrupt downstream state. 

20 

21The current latency posture is synchronous block with a tight 

22``latency_deadline_ms`` and fail-open on exceed: ``_post_authorize`` 

23honors the deadline as a local timeout floor and never raises into 

24caller code. A future fire-and-forget allow-path mode would wrap this 

25module's helpers rather than replacing them. 

26""" 

27from __future__ import annotations 

28 

29import atexit 

30import contextlib 

31import logging 

32import math 

33import os 

34import threading 

35from dataclasses import dataclass 

36from typing import Any, Literal

38import httpx

40logger = logging.getLogger(__name__) 

41 

42PrivacyTier = Literal["metadata", "embeddings", "hashed_prompts"] 

43 

44# The only privacy tier wired up today. ``embeddings`` and 

45# ``hashed_prompts`` are reserved on the public type so opt-in flags 

46# don't break compatibility when they ship, but accepting them silently 

47# now would be a bug surface (operator thinks they have embeddings when 

48# they don't). ``CloudConfig`` therefore rejects them at construction. 

49_SUPPORTED_PRIVACY_TIERS: frozenset[str] = frozenset({"metadata"}) 

50_PRIVACY_TIER_FOLLOWUP_HINT = ( 

51 "embeddings / hashed_prompts privacy tiers are not yet implemented " 

52 "and will raise at construction. Use privacy_tier='metadata' until " 

53 "client-side embedding support ships in a future release." 

54) 

55 

56# Server response envelope contracts. The MCP client consumes the same 

57# wire schema, so the two clients must agree on what "valid envelope" 

58# means or one of them will corrupt local state when the server drifts. 

59_VALID_SERVER_ACTIONS: frozenset[str] = frozenset({"allow", "reroute", "halt"}) 

60_VALID_BUDGET_PRESSURE: frozenset[str] = frozenset( 

61 {"low", "moderate", "high", "critical"} 

62) 

63_VALID_COLDSTART_SOURCES: frozenset[str] = frozenset( 

64 {"prior", "shrunk", "warm", "prior_unavailable"} 

65) 

66

68@dataclass(frozen=True) 

69class CloudConfig: 

70 """Opt-in cloud-sync configuration for ``pipeline()``. 

71 

72 Construction is the first place a customer could misconfigure 

73 cloud-sync, so we validate aggressively here: 

74 

75 - ``timeout_s`` and ``latency_deadline_ms`` must be finite and > 0. 

76 - ``privacy_tier`` outside ``metadata`` raises ``NotImplementedError`` 

77 pending future client-side embedding support. 

78 - ``api_key`` resolves at construction time: explicit kwarg wins, 

79 else ``os.environ["L6E_API_KEY"]``, else ``None``. A ``None`` key 

80 after resolution does *not* raise — it's logged at WARNING and 

81 the gate fails open immediately on every ``check()`` (so customer 

82 pipelines that briefly forget the env var degrade to local-only 

83 enforcement, not crash). 

84 """ 

85 

86 base_url: str = "https://relay.l6e.ai" 

87 api_key: str | None = None 

88 timeout_s: float = 0.250 

89 latency_deadline_ms: int = 250 

90 privacy_tier: PrivacyTier = "metadata" 

91 

92 def __post_init__(self) -> None: 

93 if not isinstance(self.base_url, str) or not self.base_url.strip(): 

94 raise ValueError("CloudConfig.base_url must be a non-empty string") 

95 timeout = float(self.timeout_s) 

96 if math.isnan(timeout) or math.isinf(timeout) or timeout <= 0: 

97 raise ValueError( 

98 "CloudConfig.timeout_s must be a finite positive number " 

99 f"(got {self.timeout_s!r})" 

100 ) 

101 if not isinstance(self.latency_deadline_ms, int) or isinstance( 

102 self.latency_deadline_ms, bool, 

103 ) or self.latency_deadline_ms <= 0: 

104 raise ValueError( 

105 "CloudConfig.latency_deadline_ms must be a positive int " 

106 f"(got {self.latency_deadline_ms!r})" 

107 ) 

108 if self.privacy_tier not in _SUPPORTED_PRIVACY_TIERS: 

109 raise NotImplementedError( 

110 f"CloudConfig.privacy_tier={self.privacy_tier!r} is not yet supported. " 

111 + _PRIVACY_TIER_FOLLOWUP_HINT 

112 ) 

113 # ``api_key`` resolution: explicit > env > None. We mutate via 

114 # ``object.__setattr__`` because the dataclass is frozen. 

115 if self.api_key is None: 

116 env_key = os.environ.get("L6E_API_KEY") 

117 if env_key: 

118 object.__setattr__(self, "api_key", env_key) 

119 if not self.api_key: 

120 logger.warning( 

121 "cloud_config_missing_api_key", 

122 extra={"base_url": self.base_url}, 

123 ) 

124 

125 @property 

126 def effective_timeout_s(self) -> float: 

127 """The HTTP timeout used per call: tighter of ``timeout_s`` and 

128 ``latency_deadline_ms``. Mirrors the MCP client's deadline-honored 

129 path so server-slow degradation has a hard local ceiling.""" 

130 deadline_seconds = self.latency_deadline_ms / 1000.0 

131 return min(float(self.timeout_s), deadline_seconds)

134# --------------------------------------------------------------------------- 

135# Shared sync httpx.Client — module-singleton with atexit cleanup. Reusing 

136# a single client across calls eliminates per-request DNS and TLS overhead. 

137# --------------------------------------------------------------------------- 

138 

139_client: httpx.Client | None = None 

140_client_lock = threading.Lock() 

141 

142 

143def _get_sync_client(timeout: float) -> httpx.Client: 

144 global _client # noqa: PLW0603 

145 if _client is not None: 

146 return _client 

147 with _client_lock: 

148 if _client is not None: 

149 return _client 

150 _client = httpx.Client(timeout=timeout) 

151 return _client 

152 

153 

154def _shutdown_client() -> None: 

155 global _client # noqa: PLW0603 

156 with _client_lock: 

157 to_close = _client 

158 _client = None 

159 if to_close is not None: 

160 # atexit must not raise; a broken client tear-down is strictly 

161 # preferable to crashing the interpreter on exit. 

162 with contextlib.suppress(Exception): 

163 to_close.close() 

164 

165 

166atexit.register(_shutdown_client) 

167 

168 

169def _reset_client() -> None: 

170 """Clear the cached client. Used by tests for isolation.""" 

171 global _client # noqa: PLW0603 

172 with _client_lock: 

173 if _client is not None: 

174 with contextlib.suppress(Exception): 

175 _client.close() 

176 _client = None

179# --------------------------------------------------------------------------- 

180# Response sanitizer — defensive, returns None on any malformed envelope. 

181# --------------------------------------------------------------------------- 

182

184def _finite_non_negative(value: object) -> float | None: 

185 """Coerce a server-supplied numeric to a sane non-negative float. 

186 

187 Returns ``None`` for NaN, inf, negative, non-numeric, or ``bool`` 

188 (since ``bool`` is an ``int`` subclass and silent True→1 promotion 

189 would mask a wire bug). 

190 """ 

191 if isinstance(value, bool) or value is None: 

192 return None 

193 if not isinstance(value, (int, float)): 

194 return None 

195 fvalue = float(value) 

196 if math.isnan(fvalue) or math.isinf(fvalue) or fvalue < 0: 

197 return None 

198 return fvalue 

199

201def _sanitize_authorize_response(resp: object) -> dict[str, Any] | None: 

202 """Validate a ``/v1/authorize`` response envelope. 

203 

204 Returns the dict on success or ``None`` on any garbage that should 

205 trigger fail-open. The validation rules match the MCP client's 

206 sanitizer field-for-field: the two clients must agree on what 

207 "valid envelope" means or one of them will corrupt local state when 

208 the server drifts. 

209 """ 

210 if not isinstance(resp, dict): 

211 return None 

212 if resp.get("action") not in _VALID_SERVER_ACTIONS: 

213 return None 

214 if _finite_non_negative(resp.get("calibrated_cost_usd")) is None: 

215 return None 

216 if _finite_non_negative(resp.get("remaining_usd")) is None: 

217 return None 

218 if resp.get("budget_pressure") not in _VALID_BUDGET_PRESSURE: 

219 return None 

220 factor_raw = resp.get("calibration_factor") 

221 if factor_raw is not None and _finite_non_negative(factor_raw) is None: 

222 return None 

223 coldstart_source = resp.get("coldstart_source") 

224 if ( 

225 coldstart_source is not None 

226 and coldstart_source not in _VALID_COLDSTART_SOURCES 

227 ): 

228 return None 

229 # Margin-tier prediction fields are optional but if present must be sane. 

230 for key in ("predicted_cost_mean_usd", "predicted_cost_p95_usd"): 

231 v = resp.get(key) 

232 if v is not None and _finite_non_negative(v) is None: 

233 return None 

234 return resp

242def _post_authorize( 

243 cfg: CloudConfig, 

244 body: dict[str, Any], 

245) -> dict[str, Any] | None: 

246 """POST to ``{cfg.base_url}/v1/authorize``. Returns sanitized dict or None. 

247 

248 Failure modes mapped to ``None`` (caller falls back to local gate): 

249 

250 - missing API key (config-time) — logged at WARNING by ``CloudConfig``, 

251 we just bail here with a stable structured log. 

252 - ``httpx.TimeoutException`` (deadline-honored or server-slow). 

253 - non-200 HTTP status. 

254 - malformed JSON body on a 200 response. 

255 - any other ``Exception`` (network, DNS, TLS, etc.). 

256 - server response that fails ``_sanitize_authorize_response``. 

257 

258 Each path emits a stable, operator-greppable log key: 

259 ``cloud_authorize_no_api_key``, ``cloud_authorize_timeout``, 

260 ``cloud_authorize_5xx``, ``cloud_authorize_bad_json``, 

261 ``cloud_authorize_invalid_envelope``, ``cloud_authorize_failed``. 

262 

263 Never raises into caller code. The iron rule is uncompromising here. 

264 """ 

265 if not cfg.api_key: 

266 logger.warning( 

267 "cloud_authorize_no_api_key", 

268 extra={"base_url": cfg.base_url}, 

269 ) 

270 return None 

271 

272 url = f"{cfg.base_url.rstrip('/')}/v1/authorize" 

273 effective_timeout = cfg.effective_timeout_s 

274 

275 try: 

276 client = _get_sync_client(effective_timeout) 

277 except Exception: 

278 # Defensive: if even client construction fails (broken httpx 

279 # install, OOM, etc.), fail-open. There's no recovery path here 

280 # except local enforcement. 

281 logger.warning("cloud_authorize_failed", exc_info=True) 

282 return None 

283 

284 try: 

285 resp = client.post( 

286 url, 

287 json=body, 

288 headers={"Authorization": f"Bearer {cfg.api_key}"}, 

289 timeout=effective_timeout, 

290 ) 

291 except httpx.TimeoutException: 

292 logger.warning( 

293 "cloud_authorize_timeout", 

294 extra={ 

295 "url": url, 

296 "effective_timeout": effective_timeout, 

297 }, 

298 ) 

299 return None 

300 except Exception: 

301 logger.warning("cloud_authorize_failed", exc_info=True) 

302 return None 

303 

304 if resp.status_code != 200: 

305 logger.warning( 

306 "cloud_authorize_5xx", 

307 extra={"status": resp.status_code, "body": resp.text[:200]}, 

308 ) 

309 return None 

310 

311 try: 

312 parsed = resp.json() 

313 except Exception: 

314 logger.warning("cloud_authorize_bad_json", exc_info=True) 

315 return None 

316 

317 sanitized = _sanitize_authorize_response(parsed) 

318 if sanitized is None: 

319 logger.warning( 

320 "cloud_authorize_invalid_envelope", 

321 extra={"keys": sorted(parsed.keys()) if isinstance(parsed, dict) else None}, 

322 ) 

323 return None 

324 return sanitized
