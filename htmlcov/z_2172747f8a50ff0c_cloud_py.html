1"""Cloud-sync configuration and ``/v1/authorize`` client for the SDK. 

2 

3Opt-in surface for the cloud-sync integration tier. A customer wires it 

4via the ``cloud=`` kwarg on ``pipeline()``; absent that kwarg, the SDK 

5behaves exactly as it did before this module existed (purely local 

6enforcement, no network calls). 

7 

8This module owns three responsibilities: 

9 

101. ``CloudConfig`` — the user-facing configuration surface. 

112. ``_post_authorize`` — the synchronous HTTP client for 

12 ``POST /v1/authorize``. Returns ``None`` on every failure path 

13 (timeout, non-2xx, malformed JSON, network exception, missing API 

14 key, garbage envelope) so the caller can fall back to local 

15 enforcement. The gate fails open, always — that's the iron rule. 

163. ``_sanitize_authorize_response`` — defensive validation of the server 

17 envelope before any field is used to drive a ``GateDecision``. NaN / 

18 negative / missing-action / unknown-pressure-label all collapse to 

19 ``None`` so a poisoned response can't quietly corrupt downstream state. 

20 

21The current latency posture is synchronous block with a tight 

22``latency_deadline_ms`` and fail-open on exceed: ``_post_authorize`` 

23honors the deadline as a local timeout floor and never raises into 

24caller code. A future fire-and-forget allow-path mode would wrap this 

25module's helpers rather than replacing them. 

26"""

44PrivacyTier = Literal["metadata", "embeddings", "hashed_prompts"] 

45 

46# Customer-supplied client-side embedder. Receives the SDK-extracted user 

47# prompts (list of strings, in message order) and returns a single 

48# ``list[float]`` to attach as ``request_embedding`` on the cloud 

49# authorize body. Bring-your-own — l6e does not prescribe a model. The 

50# customer owns pooling (mean / first-only / concat-then-embed) so they 

51# can match whatever their privacy-reviewed embedder produces. 

52Embedder = Callable[[list[str]], list[float]] 

53 

54# Privacy tiers wired up today. ``hashed_prompts`` is reserved on the 

55# public type so opt-in flags don't break compatibility when L6E-98 

56# ships, but accepting it silently now would be a bug surface (operator 

57# thinks they have hashed prompts when they don't). ``CloudConfig`` 

58# therefore rejects it at construction. 

59_SUPPORTED_PRIVACY_TIERS: frozenset[str] = frozenset({"metadata", "embeddings"}) 

60_PRIVACY_TIER_FOLLOWUP_HINT = ( 

61 "hashed_prompts privacy tier is not yet implemented and will raise " 

62 "at construction. Use privacy_tier='metadata' or " 

63 "privacy_tier='embeddings' until LSH/MinHash support ships (L6E-98)." 

64) 

65 

66# Server-side cap on embedding dimensionality (mirrored from 

67# ``hosted-edge/src/relay/routers/authorize.py:_MAX_EMBEDDING_DIM``). 

68# Validating client-side keeps a single ``cloud_embedding_failed`` 

69# fail-surface — the alternative is letting the server 400 and 

70# fail-opening through the ``cloud_authorize_5xx`` path, which conflates 

71# "embedder produced garbage" with "server is unhealthy" in operator 

72# dashboards. Iron-rule extension: client-side validation is the right 

73# place to catch bring-your-own-embedder misconfiguration. 

74_MAX_EMBEDDING_DIM = 4096 

75 

76# Server response envelope contracts. The MCP client consumes the same 

77# wire schema, so the two clients must agree on what "valid envelope" 

78# means or one of them will corrupt local state when the server drifts. 

79_VALID_SERVER_ACTIONS: frozenset[str] = frozenset({"allow", "reroute", "halt"}) 

80_VALID_BUDGET_PRESSURE: frozenset[str] = frozenset( 

81 {"low", "moderate", "high", "critical"} 

82) 

83_VALID_COLDSTART_SOURCES: frozenset[str] = frozenset( 

84 {"prior", "shrunk", "warm", "prior_unavailable"} 

85)

88@dataclass(frozen=True) 

89class CloudConfig: 

90 """Opt-in cloud-sync configuration for ``pipeline()``. 

91 

92 Construction is the first place a customer could misconfigure 

93 cloud-sync, so we validate aggressively here: 

94 

95 - ``timeout_s`` and ``latency_deadline_ms`` must be finite and > 0. 

96 - ``privacy_tier="hashed_prompts"`` raises ``NotImplementedError`` 

97 pending L6E-98. 

98 - ``privacy_tier="embeddings"`` requires ``embedder`` to be set; 

99 missing ``embedder`` raises ``ValueError`` at construction. This 

100 is a misconfig, not a runtime fail-open: a customer who declared 

101 "I want embeddings tier" but forgot the embedder would otherwise 

102 silently degrade to metadata on every call, defeating the purpose 

103 of the opt-in. Catch it loudly at load time. 

104 - ``api_key`` resolves at construction time: explicit kwarg wins, 

105 else ``os.environ["L6E_API_KEY"]``, else ``None``. A ``None`` key 

106 after resolution does *not* raise — it's logged at WARNING and 

107 the gate fails open immediately on every ``check()`` (so customer 

108 pipelines that briefly forget the env var degrade to local-only 

109 enforcement, not crash). 

110 """ 

111 

112 base_url: str = "https://relay.l6e.ai" 

113 api_key: str | None = None 

114 timeout_s: float = 0.250 

115 latency_deadline_ms: int = 250 

116 privacy_tier: PrivacyTier = "metadata" 

117 # Customer-supplied client-side embedder. Required when 

118 # ``privacy_tier="embeddings"``; ignored otherwise. Setting it under 

119 # ``metadata`` tier is allowed (and silent) so customers can flip 

120 # tiers via env without conditionally removing the embedder. 

121 embedder: Embedder | None = None 

122 

123 def __post_init__(self) -> None: 

124 if not isinstance(self.base_url, str) or not self.base_url.strip(): 

125 raise ValueError("CloudConfig.base_url must be a non-empty string") 

126 timeout = float(self.timeout_s) 

127 if math.isnan(timeout) or math.isinf(timeout) or timeout <= 0: 

128 raise ValueError( 

129 "CloudConfig.timeout_s must be a finite positive number " 

130 f"(got {self.timeout_s!r})" 

131 ) 

132 if not isinstance(self.latency_deadline_ms, int) or isinstance( 

133 self.latency_deadline_ms, bool, 

134 ) or self.latency_deadline_ms <= 0: 

135 raise ValueError( 

136 "CloudConfig.latency_deadline_ms must be a positive int " 

137 f"(got {self.latency_deadline_ms!r})" 

138 ) 

139 if self.privacy_tier not in _SUPPORTED_PRIVACY_TIERS: 

140 raise NotImplementedError( 

141 f"CloudConfig.privacy_tier={self.privacy_tier!r} is not yet supported. " 

142 + _PRIVACY_TIER_FOLLOWUP_HINT 

143 ) 

144 if self.privacy_tier == "embeddings" and self.embedder is None: 

145 raise ValueError( 

146 "CloudConfig.privacy_tier='embeddings' requires an " 

147 "embedder. Pass embedder=<callable[[list[str]], list[float]]> " 

148 "or set privacy_tier='metadata'. Bring-your-own; l6e does " 

149 "not prescribe an embedding model." 

150 ) 

151 if self.embedder is not None and not callable(self.embedder): 

152 raise ValueError( 

153 "CloudConfig.embedder must be callable " 

154 f"(got {type(self.embedder).__name__})" 

155 ) 

156 # ``api_key`` resolution: explicit > env > None. We mutate via 

157 # ``object.__setattr__`` because the dataclass is frozen. 

158 if self.api_key is None: 

159 env_key = os.environ.get("L6E_API_KEY") 

160 if env_key: 

161 object.__setattr__(self, "api_key", env_key) 

162 if not self.api_key: 

163 logger.warning( 

164 "cloud_config_missing_api_key", 

165 extra={"base_url": self.base_url}, 

166 )

168 @property 

169 def effective_timeout_s(self) -> float: 

170 """The HTTP timeout used per call: tighter of ``timeout_s`` and 

171 ``latency_deadline_ms``. Mirrors the MCP client's deadline-honored 

172 path so server-slow degradation has a hard local ceiling.""" 

173 deadline_seconds = self.latency_deadline_ms / 1000.0 

174 return min(float(self.timeout_s), deadline_seconds)

178# Shared sync httpx.Client — module-singleton with atexit cleanup. Reusing 

179# a single client across calls eliminates per-request DNS and TLS overhead. 

180# ---------------------------------------------------------------------------

182_client: httpx.Client | None = None 

183_client_lock = threading.Lock()

186def _get_sync_client(timeout: float) -> httpx.Client: 

187 global _client # noqa: PLW0603 

188 if _client is not None: 

189 return _client 

190 with _client_lock: 

191 if _client is not None: 

192 return _client 

193 _client = httpx.Client(timeout=timeout) 

194 return _client

197def _shutdown_client() -> None: 

198 global _client # noqa: PLW0603 

199 with _client_lock: 

200 to_close = _client 

201 _client = None 

202 if to_close is not None: 

203 # atexit must not raise; a broken client tear-down is strictly 

204 # preferable to crashing the interpreter on exit. 

205 with contextlib.suppress(Exception): 

206 to_close.close() 

207 

208 

209atexit.register(_shutdown_client)

212def _reset_client() -> None: 

213 """Clear the cached client. Used by tests for isolation.""" 

214 global _client # noqa: PLW0603 

215 with _client_lock: 

216 if _client is not None: 

217 with contextlib.suppress(Exception): 

218 _client.close() 

219 _client = None

223# Embedder invocation — fail-open to metadata-tier behavior on any failure. 

224# ---------------------------------------------------------------------------

227def _validate_embedding(value: object) -> list[float] | None: 

228 """Defensive shape check matching the server-side 

229 ``_require_embedding`` contract (``hosted-edge/.../authorize.py``). 

230 

231 Returns the coerced ``list[float]`` on success or ``None`` on any 

232 structural problem. Catching client-side keeps a single 

233 ``cloud_embedding_failed`` fail-surface — the alternative is letting 

234 the server 400 and routing through ``cloud_authorize_5xx``, which 

235 conflates "embedder produced garbage" with "server is unhealthy" in 

236 operator dashboards. 

237 

238 Rejects: non-list, empty list, > ``_MAX_EMBEDDING_DIM`` dims, 

239 elements that aren't ``Real`` (or are ``bool``, since ``True/False`` 

240 silently coerce to 1/0 and mask wire bugs), NaN, inf. 

241 """ 

242 if not isinstance(value, list) or not value: 

243 return None 

244 if len(value) > _MAX_EMBEDDING_DIM: 

245 return None 

246 coerced: list[float] = [] 

247 for item in value: 

248 if isinstance(item, bool) or not isinstance(item, Real): 

249 return None 

250 fitem = float(item) 

251 if math.isnan(fitem) or math.isinf(fitem): 

252 return None 

253 coerced.append(fitem) 

254 return coerced

257def _safe_embed( 

258 embedder: Embedder, prompts: list[str], 

259) -> list[float] | None: 

260 """Invoke a customer-supplied embedder and validate the result. 

261 

262 Returns the validated ``list[float]`` on success, or ``None`` on any 

263 failure mode (embedder raised, returned non-list, returned empty, 

264 too many dims, NaN / inf, non-numeric elements). Never raises into 

265 caller code — the iron-rule extension for the embeddings tier: 

266 any embedder failure collapses to metadata-tier behavior (the 

267 ``request_embedding`` field is omitted from the cloud body), the 

268 cloud call still drives the decision. 

269 

270 A single stable log key — ``cloud_embedding_failed`` — fires on every 

271 failure. Operators pivot dashboards on this one tag rather than on a 

272 family of subkeys, mirroring how ``_post_authorize`` collapses many 

273 network failure modes onto a small set of greppable strings. 

274 """ 

275 try: 

276 result = embedder(prompts) 

277 except Exception: 

278 logger.warning("cloud_embedding_failed", exc_info=True) 

279 return None 

280 validated = _validate_embedding(result) 

281 if validated is None: 

282 # Distinguish "embedder ran but returned garbage" from "embedder 

283 # raised" via the structured ``reason`` extra without a separate 

284 # log key. The two failure modes route through the same 

285 # fail-open path so they share the same operator surface. 

286 logger.warning( 

287 "cloud_embedding_failed", 

288 extra={"reason": "invalid_embedding_shape"}, 

289 ) 

290 return None 

291 return validated

295# Response sanitizer — defensive, returns None on any malformed envelope.

299def _finite_non_negative(value: object) -> float | None: 

300 """Coerce a server-supplied numeric to a sane non-negative float. 

301 

302 Returns ``None`` for NaN, inf, negative, non-numeric, or ``bool`` 

303 (since ``bool`` is an ``int`` subclass and silent True→1 promotion 

304 would mask a wire bug). 

305 """ 

306 if isinstance(value, bool) or value is None: 

307 return None 

308 if not isinstance(value, (int, float)): 

309 return None 

310 fvalue = float(value) 

311 if math.isnan(fvalue) or math.isinf(fvalue) or fvalue < 0: 

312 return None 

313 return fvalue

316def _sanitize_authorize_response(resp: object) -> dict[str, Any] | None: 

317 """Validate a ``/v1/authorize`` response envelope. 

318 

319 Returns the dict on success or ``None`` on any garbage that should 

320 trigger fail-open. The validation rules match the MCP client's 

321 sanitizer field-for-field: the two clients must agree on what 

322 "valid envelope" means or one of them will corrupt local state when 

323 the server drifts. 

324 """ 

325 if not isinstance(resp, dict): 

326 return None 

327 if resp.get("action") not in _VALID_SERVER_ACTIONS: 

328 return None 

329 if _finite_non_negative(resp.get("calibrated_cost_usd")) is None: 

330 return None 

331 if _finite_non_negative(resp.get("remaining_usd")) is None: 

332 return None 

333 if resp.get("budget_pressure") not in _VALID_BUDGET_PRESSURE: 

334 return None 

335 factor_raw = resp.get("calibration_factor") 

336 if factor_raw is not None and _finite_non_negative(factor_raw) is None: 

337 return None 

338 coldstart_source = resp.get("coldstart_source") 

339 if ( 

340 coldstart_source is not None 

341 and coldstart_source not in _VALID_COLDSTART_SOURCES 

342 ): 

343 return None 

344 # Margin-tier prediction fields are optional but if present must be sane. 

345 for key in ("predicted_cost_mean_usd", "predicted_cost_p95_usd"): 

346 v = resp.get(key) 

347 if v is not None and _finite_non_negative(v) is None: 

348 return None 

349 return resp

357def _post_authorize( 

358 cfg: CloudConfig, 

359 body: dict[str, Any], 

360) -> dict[str, Any] | None: 

361 """POST to ``{cfg.base_url}/v1/authorize``. Returns sanitized dict or None. 

362 

363 Failure modes mapped to ``None`` (caller falls back to local gate): 

364 

365 - missing API key (config-time) — logged at WARNING by ``CloudConfig``, 

366 we just bail here with a stable structured log. 

367 - ``httpx.TimeoutException`` (deadline-honored or server-slow). 

368 - non-200 HTTP status. 

369 - malformed JSON body on a 200 response. 

370 - any other ``Exception`` (network, DNS, TLS, etc.). 

371 - server response that fails ``_sanitize_authorize_response``. 

372 

373 Each path emits a stable, operator-greppable log key: 

374 ``cloud_authorize_no_api_key``, ``cloud_authorize_timeout``, 

375 ``cloud_authorize_5xx``, ``cloud_authorize_bad_json``, 

376 ``cloud_authorize_invalid_envelope``, ``cloud_authorize_failed``. 

377 

378 Never raises into caller code. The iron rule is uncompromising here. 

379 """ 

380 if not cfg.api_key: 

381 logger.warning( 

382 "cloud_authorize_no_api_key", 

383 extra={"base_url": cfg.base_url}, 

384 ) 

385 return None 

386 

387 url = f"{cfg.base_url.rstrip('/')}/v1/authorize" 

388 effective_timeout = cfg.effective_timeout_s 

389 

390 try: 

391 client = _get_sync_client(effective_timeout) 

392 except Exception: 

393 # Defensive: if even client construction fails (broken httpx 

394 # install, OOM, etc.), fail-open. There's no recovery path here 

395 # except local enforcement. 

396 logger.warning("cloud_authorize_failed", exc_info=True) 

397 return None 

398 

399 try: 

400 resp = client.post( 

401 url, 

402 json=body, 

403 headers={"Authorization": f"Bearer {cfg.api_key}"}, 

404 timeout=effective_timeout, 

405 ) 

406 except httpx.TimeoutException: 

407 logger.warning( 

408 "cloud_authorize_timeout", 

409 extra={ 

410 "url": url, 

411 "effective_timeout": effective_timeout, 

412 }, 

413 ) 

414 return None 

415 except Exception: 

416 logger.warning("cloud_authorize_failed", exc_info=True) 

417 return None 

418 

419 if resp.status_code != 200: 

420 logger.warning( 

421 "cloud_authorize_5xx", 

422 extra={"status": resp.status_code, "body": resp.text[:200]}, 

423 ) 

424 return None 

425 

426 try: 

427 parsed = resp.json() 

428 except Exception: 

429 logger.warning("cloud_authorize_bad_json", exc_info=True) 

430 return None 

431 

432 sanitized = _sanitize_authorize_response(parsed) 

433 if sanitized is None: 

434 logger.warning( 

435 "cloud_authorize_invalid_envelope", 

436 extra={"keys": sorted(parsed.keys()) if isinstance(parsed, dict) else None}, 

437 ) 

438 return None 

439 return sanitized
