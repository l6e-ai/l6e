53@dataclass(frozen=True) 

54class PipelinePolicy: 

55 """OSS: hand-authored. Pro: generated by profiling agents from run history. 

56 

57 Construction-time validation (see ``__post_init__``) rejects obvious 

58 misconfigurations — zero / negative budgets, out-of-range 

59 ``reroute_threshold``, negative pricing floors. This is the 

60 iron-rule "validation at upload time" for the SDK: a customer who 

61 writes ``PipelinePolicy(budget=0)`` gets a ``ValueError`` at load, 

62 not a silent fail-closed gate at runtime. 

63 """ 

64 

65 budget: float 

66 budget_mode: BudgetMode = BudgetMode.HALT 

67 on_budget_exceeded: OnBudgetExceeded = OnBudgetExceeded.RAISE 

68 fallback_result: str | None = None 

69 latency_sla: float | None = None # not enforced in v0.1; reserved for v0.2 

70 reroute_threshold: float = 0.8 

71 unknown_model_cost_per_1k_tokens: float = 0.01 

72 unknown_model_pricing_mode: UnknownModelPricingMode = ( 

73 UnknownModelPricingMode.WARN_ONLY 

74 ) 

75 stage_routing: dict[str, StageRoutingHint] = field(default_factory=dict) 

76 stage_overrides: dict[str, BudgetMode] = field(default_factory=dict) 

77 

78 def __post_init__(self) -> None: 

79 self._validate() 

80 

81 def _validate(self) -> None: 

82 """Sanity-check the policy. Raises ``ValueError`` on misconfig. 

83 

84 Intentionally strict: the OSS gate must be predictable. A policy 

85 that silently widens to allow-everything is worse than one that 

86 rejects at construction. 

87 """ 

88 if not isinstance(self.budget, Real) or isinstance(self.budget, bool): 

89 raise ValueError( 

90 f"PipelinePolicy.budget must be a number, got {type(self.budget).__name__}" 

91 ) 

92 # NaN / inf / negative — each a way the gate silently breaks. 

93 budget_val = float(self.budget) 

94 if math.isnan(budget_val) or math.isinf(budget_val): 

95 raise ValueError("PipelinePolicy.budget must be finite") 

96 if budget_val < 0: 

97 raise ValueError( 

98 f"PipelinePolicy.budget must be >= 0 (got {budget_val}). " 

99 "A negative budget halts every call and is almost always a misconfig." 

100 ) 

101 

102 if not isinstance(self.reroute_threshold, Real) or isinstance( 

103 self.reroute_threshold, bool, 

104 ): 

105 raise ValueError( 

106 "PipelinePolicy.reroute_threshold must be a number, got " 

107 f"{type(self.reroute_threshold).__name__}" 

108 ) 

109 threshold_val = float(self.reroute_threshold) 

110 if math.isnan(threshold_val) or math.isinf(threshold_val): 

111 raise ValueError("PipelinePolicy.reroute_threshold must be finite") 

112 if not (0.0 <= threshold_val <= 1.0): 

113 raise ValueError( 

114 "PipelinePolicy.reroute_threshold must be in [0.0, 1.0] " 

115 f"(got {threshold_val})." 

116 ) 

117 

118 unknown_cost = float(self.unknown_model_cost_per_1k_tokens) 

119 if math.isnan(unknown_cost) or math.isinf(unknown_cost) or unknown_cost < 0: 

120 raise ValueError( 

121 "PipelinePolicy.unknown_model_cost_per_1k_tokens must be a " 

122 f"finite, non-negative number (got {unknown_cost})." 

123 ) 

124 

125 if self.latency_sla is not None: 

126 sla_val = float(self.latency_sla) 

127 if math.isnan(sla_val) or math.isinf(sla_val) or sla_val < 0: 

128 raise ValueError( 

129 "PipelinePolicy.latency_sla must be a finite, non-negative " 

130 f"number or None (got {sla_val})." 

131 )

133 @classmethod 

134 def from_toml(cls, path: Path) -> PipelinePolicy: 

135 """Load a PipelinePolicy from a TOML file. 

136 

137 Expected structure: 

138 [policy] — budget and top-level settings 

139 [stage_routing] — stage name → StageRoutingHint value 

140 [stage_overrides] — stage name → BudgetMode value 

141 """ 

142 with open(path, "rb") as f: 

143 data = tomllib.load(f) 

144 

145 policy_section = data.get("policy", {}) 

146 stage_routing_raw = data.get("stage_routing", {}) 

147 stage_overrides_raw = data.get("stage_overrides", {}) 

148 

149 stage_routing = {k: StageRoutingHint(v) for k, v in stage_routing_raw.items()} 

150 stage_overrides = {k: BudgetMode(v) for k, v in stage_overrides_raw.items()} 

151 

152 return cls( 

153 budget=float(policy_section["budget"]), 

154 budget_mode=BudgetMode(policy_section.get("budget_mode", BudgetMode.HALT)), 

155 on_budget_exceeded=OnBudgetExceeded( 

156 policy_section.get("on_budget_exceeded", OnBudgetExceeded.RAISE) 

157 ), 

158 fallback_result=policy_section.get("fallback_result"), 

159 latency_sla=policy_section.get("latency_sla"), 

160 reroute_threshold=float( 

161 policy_section.get("reroute_threshold", 0.8) 

162 ), 

163 unknown_model_cost_per_1k_tokens=float( 

164 policy_section.get("unknown_model_cost_per_1k_tokens", 0.01) 

165 ), 

166 unknown_model_pricing_mode=UnknownModelPricingMode( 

167 policy_section.get( 

168 "unknown_model_pricing_mode", 

169 UnknownModelPricingMode.WARN_ONLY, 

170 ) 

171 ), 

172 stage_routing=stage_routing, 

173 stage_overrides=stage_overrides, 

174 )
